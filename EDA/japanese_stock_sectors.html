_code = []
for codelist in codelists_1:
    tmp = codelist
    _code.append(tmp)

for codelist in codelists_2:
    tmp = codelist + '.JP'   
    _code.append(tmp)
# 開始・終了日の設定
start = datetime.date.today() - datetime.timedelta(days=180)

df = web.DataReader(_code, 'stooq', start, end)['Close']
